
EchoWithColor(COLOR GREEN "-- Configure ${CMAKE_CURRENT_LIST_DIR}")

    ${PROJECT_TEST_SRC_DIR}/*.cpp

    set_property(TARGET atframe_utils_unit_test PROPERTY FOLDER "atframework")
